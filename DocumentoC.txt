package practica2;

public class ejer1 {
		int h,m,s;
		 public ejer1(int h, int m, int s) 
		 {
		  this.h = h;
		  this.m = m;
		  this.s = s;
		 }
		public int getH() {
			return h;
		}
		public int getM() {
			return m;
		}
		public int getS() {
			return s;
		}
		
		public String toString() {
			return "La hora correcta es: "+horas();
		}
		public String horas(){
			s=s+(m*60)+(h*3600);
			m=0;h=0;
			while (s>=60){
				s=s-60;
				m++;
			}
			while (m>=60){
				m=m-60;
				h++;
			}
			return h+":"+m+":"+s;
			return h+":"+m+":"+s;
		}

	}
